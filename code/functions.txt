1. Заполняет таблицу all_users (void) -- add_new_user(user_id_ integer, is_admin_ boolean)
2. Заполняет таблицу client (void) -- add_new_client(client_id_ integer, client_name_ char, client_email_ char)
3. Заполняет таблицу items (void) -- add_new_item(item_name_ char, item_space_ integer, owner_ integer)
4. Заполняет таблицу skladi (void) -- add_new_sklad(owner_id_ integer, free_space_ integer, total_space_ integer, price_for_one_month_ integer)
5. Заполняет таблицу contracts (void) -- add_new_contract(sklad_id_ integer, client_id_ integer, rent_duration_ integer)
6. Заполняет таблицу skladi_owner (void) -- add_new_owner(owner_id_ integer, owner_name_ char, owner_email_ char)
7. Считает общее место, которое занимают вещи клиента (integer) -- general_item_space(client_id_ integer)
8. Выводит доступные клиенту склады, на которых хватит места (?) -- available_sklads(client_id integer)
9. Выводит информацию о вещах клиента (?) -- info_about_items(client_id_ integer)
10. Выводит информацию о складах овнеру (?) -- info_about_owner_sklads(user_id_ integer)
11. Выводит клиенту все его контракты (?) -- info_about_client_contracts(client_id_ integer)
12. Выводит овнеру все его контракты (?) -- info_about_owner_contracts(owner_id_ integer)
13. Проверяет админ пльзователь или нет (boolean) -- is_user_admin(user_id_ integer)
14. Удаление овнером склада -- (void) del_sklad(sklad_id_ integer)

15. Обновляет счётчик id склада (void) -- reset_skladi_inc()
16. Обновляет счётчик id контрактов (void) -- reset_contracts_inc()
17. Обновляет счётчик id вещей (void) -- reset_item_inc()
18. Проверяет, является ли юзер клиентом (boolean) -- is_user_client(client_id_ int)
19. Проверяет, является ли юзер овнером (boolean) -- is_user_owner(user_id int)
20. Считает общий спейс вещей клиента и обновляет свободное место на складе (void) -- calculate_free_space(client_id_ integer, sklad_id_ integer)
